# /bundle folder:

ctrlp, emmet-vim, supertab, random-colorscheme-picker


# /ftplugin/ruby.vim:

setlocal tabstop=2
setlocal shiftwidth=2


# .vimrc:

execute pathogen#infect()

set nocompatible

syntax enable

filetype plugin on

" Hide temporary buffer files in source tree
let g:netrw_list_hide='.*\.swp$'

" Show line numbers
set number

" Enable searching
:set incsearch
:set hlsearch

filetype plugin indent on
" Better indenting
set cindent
" show existing tab with 4 spaces width
set tabstop=4
" when indenting with '>', use 4 spaces width
set shiftwidth=4
" On pressing tab, insert 4 spaces
set expandtab

" Set cursor position and switch tabs with mouse
set mouse=a

" " Normal mode key bindings

" Map Ctrl+a to change color schemes
map <C-a> <esc>:CSnext<CR>


" " Insert mode key bindings

" Map Alt+b and Alt+e to skip around in insert mode
imap ∫ <C-o>b
imap √ <C-o>e
" Map 'jj' to escape key
imap jj <esc>
" Alt+y followed by space for Emmett snippets
imap ˆ <C-y>,
" Skip to the end of the line from insert mode with Alt+a
imap å <C-o>A
" Drop inside quotes for function x(""); with Alt+<
imap ≤ <C-[>hhi
" Drop inside parens for function x(); with Alt+>
imap ≥ <C-[>hi
